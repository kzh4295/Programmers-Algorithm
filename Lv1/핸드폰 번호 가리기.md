```javascript
<!-- 나의 풀이 -->
function solution(phone_number) {
    let answer = '';
    let len = phone_number.length;
    
    for(let i = 0; i < len-4; i++){
            answer += '*'
    }
    
    answer += phone_number.slice(-4)
    
    return answer;
}

```

```javascript
<!-- 타인의 풀이 -->
function solution(phone_number) {
  return phone_number.replace(/\d(?=\d{4})/g, "*");
}
```
