```javascript
function solution(arr) {
    let answer = [];
    let minNum = Math.min(...arr);
    
    if(arr.length <=1){
        answer.push(-1);
    } else{

    return answer;
}
```

```javascript
<!-- 타인의 풀이 -->
function solution(arr) {
    const min = Math.min(...arr);
    return arr.length !== 1 ? arr.filter(i => i !== min) : [-1]
}
```
