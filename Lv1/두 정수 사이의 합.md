function solution(a, b) {
    let answer = 0;
    let arr  = [];
    if(a > b){
        for(let i = b; i < a +1; i++){
            arr.push(i);
            answer= arr.reduce((acc, ele)=>{
               return acc += ele;
            }, 0)
        }
    } else if(a === b){
        return answer = a; 
    } else{
        for(let i = a; i <b +1; i++){
            arr.push(i)
            answer = arr.reduce((acc, ele)=>{
                return acc += ele;
            },0)
        }
    }

    return answer;
}
