```javascript
<!-- 나의 풀이 -->
function solution(numbers) {
    let answer;
    answer = 45 - numbers.reduce((acc, ele)=> acc +=ele, 0)
    return answer;
}
```

```javascript
<!-- 타인의 풀이 -->
function solution(numbers) {
    let answer = 0;

    for(let i = 0; i <= 9; i++) {
        if(!numbers.includes(i)) answer += i;
    }

    return answer;
}

```
